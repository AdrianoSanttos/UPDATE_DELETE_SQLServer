-- Alterando dados de uma coluna
USE PEDIDOS;

UPDATE EMPREGADOS
SET SALARIO = SALARIO * 1.2;
-- OU
UPDATE EMPREGADOS
SET SALARIO *= 1.2;

-- codigo 5
UPDATE EMPREGADOS
SET NUM_DEPEND = NUM_DEPEND + 2
WHERE CODFUN = 5;
-- OU
UPDATE EMPREGADOS
SET NUM_DEPEND += 2
WHERE CODFUN = 5;

    BAIRRO   = 'CERQUEIRA CESAR',
WHERE CODCLI = 5;
